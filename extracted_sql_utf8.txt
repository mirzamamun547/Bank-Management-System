CREATE OR REPLACE FUNCTION FIND_ACTIVE_ACCOUNT(
    p_account_number IN VARCHAR2
) RETURN SYS_REFCURSOR
IS
    v_result SYS_REFCURSOR;
BEGIN
    OPEN v_result FOR
        SELECT
            a.id            AS account_id,
            a.account_number,
            a.balance,
            a.user_id,
            u.full_name,
            u.nid
        FROM accounts a
        JOIN USERS u ON a.user_id = u.id
        WHERE a.account_number = p_account_number
        AND   a.status         = 'Active';

    RETURN v_result;

    
EXCEPTION
    WHEN OTHERS THEN
        RAISE_APPLICATION_ERROR(-20001, 'FIND_ACTIVE_ACCOUNT Error: ' || SQLERRM);
END;
